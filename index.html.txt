    // плавний скрол та підсвітка TOC
    document.querySelectorAll('#toc a').forEach(a=>{
      a.addEventListener('click', e=>{
        e.preventDefault();
        const id = a.getAttribute('href').slice(1);
        const el = document.getElementById(id);
        if(el) el.scrollIntoView({behavior:'smooth', block:'start'});
        history.replaceState(null,null,'#'+id);
      });
    });

    // підсвітка активного розділу при скролі
    const tocLinks = Array.from(document.querySelectorAll('#toc a'));
    const sections = tocLinks.map(l => document.getElementById(l.getAttribute('href').slice(1)));
    function onScroll(){
      const pos = window.scrollY + 120;
      let activeIndex = 0;              sections.forEach((s,i)=>{
        if(s && s.offsetTop <= pos) activeIndex = i;
      });
      tocLinks.forEach((l,i)=> l.style.opacity = (i===activeIndex? '1':'0.6'));
    }
    window.addEventListener('scroll', onScroll);
    onScroll();
  </script>
